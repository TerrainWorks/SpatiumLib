149  Point3 projectPoint(const Point3 &point) const override
150  {
151  // Transform point to sphere space
152  Point3 transPoint = m_ellipsoidalTransformationInverse * point;
153 
154  // Project point on unit sphere
155  Sphere sphere(Point3(0,0,0), 1);
156  Point3 transProjection = sphere.projectPoint(transPoint);
157 
158  // Transform projected point to ellipsoid space
159  return m_ellipsoidalTransformation * transProjection;
160  }

168  bool intersectLine(const Point3 &origin,
169  const Vector3 &direction,
170  Point3 &intersection) const override
171  {
172  // Transform line origin and direction to sphere space
173  Point3 transOrigin = m_ellipsoidalTransformationInverse * origin;
174  Vector3 transDirection = m_ellipsoidalTransformationInverse * direction;
175 
176  // Intersect line with unit sphere
177  Sphere sphere(Point3(0,0,0), 1);
178  Point3 transIntersection;
179  if (sphere.intersectLine(transOrigin, transDirection, transIntersection))
180  {
181  // Transform intersection to ellipsoid space
182  intersection = m_ellipsoidalTransformation * transIntersection;
183  return true;
184  }
185  else
186  {
187  return false;
188  }
189  }

195  Vector3 surfaceNormal(const Point3 &point) const
196  {
197  // Ellipsoid = x^2 / a^2 + y^2 / b^2 + z^2 / c^2 = 1
198  // Normal = (2x / a^2, 2y / b^2, 2z / c^2)
199  // = (x / a^2, y / b^2, z / c^2)
200  Vector3 normal(point.x() / (m_axisX*m_axisX),
201  point.y() / (m_axisY*m_axisY),
202  point.z() / (m_axisZ*m_axisZ));
203  normal.normalize();
204  return normal;
205  }
