175  static Matrix4x4 rotationAround(const Vector3 &axis, double angle)
176  {
177  Vector3 axisNormalized = axis.normalized();
178  double q0 = cos(angle/2);
179  double q1 = sin(angle/2) * axisNormalized.x();
180  double q2 = sin(angle/2) * axisNormalized.y();
181  double q3 = sin(angle/2) * axisNormalized.z();
182 
183  Matrix4x4 result;
184  result(0,0) = q0*q0 + q1*q1 - q2*q2 - q3*q3;
185  result(0,1) = 2*(q1*q2 - q0*q3);
186  result(0,2) = 2*(q1*q3 + q0*q2);
187  result(1,0) = 2*(q2*q1 + q0*q3);
188  result(1,1) = (q0*q0 - q1*q1 + q2*q2 - q3*q3);
189  result(1,2) = 2*(q2*q3 - q0*q1);
190  result(2,0) = 2*(q3*q1 - q0*q2);
191  result(2,1) = 2*(q3*q2 + q0*q1);
192  result(2,2) = (q0*q0 - q1*q1 - q2*q2 + q3*q3);
193  return result;
194  }
