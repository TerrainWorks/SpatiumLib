84  Point3 spheroidalToCartesianCoordinates(double longitude, double latitude,
85  double height) const
86  {
87  // http://www.navipedia.net/index.php/
88  // Ellipsoidal_and_Cartesian_Coordinates_Conversion
89  double f = 1 - m_axisZ / m_axisX;
90  double e2 = 2 * f - f*f;
91  double N = m_axisX / sqrt(1 - e2 * pow(sin(latitude), 2) );
92  double x = (N + height) * cos(latitude) * cos(longitude);
93  double y = (N + height) * cos(latitude) * sin(longitude);
94  double z = ((1 - e2) * N + height) * sin(latitude);
95  return Point3(x, y, z);
96  }

109  Point3 cartesianToSpheroidalCoordinates(double x, double y, double z,
110  double precision = 0) const
111  {
112  // http://www.navipedia.net/index.php/
113  // Ellipsoidal_and_Cartesian_Coordinates_Conversion
114 
115  // Longitude
116  double longitude = atan2(y,x);
117 
118  // Latitude
119  double f = 1 - m_axisZ / m_axisX;
120  double e2 = 2 * f - f*f;
121  double p = sqrt(x*x + y*y);
122  double latitude = atan(z / ((1 - e2) * p));
123 
124  // Height
125  double N = m_axisX / sqrt(1 - e2 * pow(sin(latitude), 2));
126  double height = (p / cos(latitude)) - N;
127 
128  // Note: latitude and height could be improved iterativly, but doesn't
129  // seem necessarily
130 
131  return Point3(longitude, latitude, height);
132  }
