196  Vector3 surfaceNormal(const Point3 &point) const
197  {
198  // Ellipsoid = x^2 / a^2 + y^2 / b^2 + z^2 / c^2 = 1
199  // Normal = Derivative
200  // = (2x / a^2, 2y / b^2, 2z / c^2)

217  Point3 pointOnHorizon(const Point3 &point, double direction, bool poleLimit = false)
218  {
219  // Convert direction to X and Y
220  double directionX = cos(direction); // Direction longitudinal
221  double directionY = sin(direction); // Direction latitudinal
222 
223  // Transform point to unit sphere space
224  Point3 transPoint = m_ellipsoidalTransformationInverse * point;
225 
226  // Unit sphere
227  Sphere sphere(1);
228 
229  // Project point on unit sphere
230  Point3 pointOnSphere = sphere.projectPoint(transPoint);
231 
232  // Compute distance to horizon on unit sphere
233  double horizonDistance =
234  sphere.distanceToHorizon(
235  pointOnSphere.distanceTo(transPoint));
236 
237  // Note: Convert distance to angular distance:
238  // angular = distance / radius.
239  // radius = 1
240  // angular = distance
241  double angularDistance = horizonDistance;
242 
243  // Convert to spherical coordinates
244  Point3 sphericalCoordinates =
245  sphere.cartesianToSphericalCoordinates(pointOnSphere.x(),
246  pointOnSphere.y(),
247  pointOnSphere.z());
248 
249  // Limit Y (latitude) near the poles if requested
250  double pointOnHorizonSphericalY =
251  sphericalCoordinates.y() + directionY * angularDistance;
252  if (poleLimit)
253  {
254  if (pointOnHorizonSphericalY > Math::PI/2)
255  {
256  pointOnHorizonSphericalY = Math::PI/2;
257  }
258  else if (pointOnHorizonSphericalY < -Math::PI/2)
259  {
260  pointOnHorizonSphericalY = -Math::PI/2;
261  }
262  }
263 
264  // Add angle towards direction
265  Point3 pointOnHorizonSpherical(
266  sphericalCoordinates.x() + directionX * angularDistance,
267  pointOnHorizonSphericalY,
268  sphericalCoordinates.z());
269 
270  // Convert spherical to cartesian coordinates
271  Point3 pointOnHorizon =
272  sphere.sphericalToCartesianCoordinates(pointOnHorizonSpherical.x(),
273  pointOnHorizonSpherical.y(),
274  pointOnHorizonSpherical.z());
275 
276  // Transform point on horizon to ellipsoid space
277  return m_ellipsoidalTransformation * pointOnHorizon;
278  }
